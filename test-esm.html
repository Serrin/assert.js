assert.js MIN test page

/* simple export */

import { assert } from "./assert.esm.js";
window.assert1 = assert;

alert(assert1(true));
// return true

alert(assert1(true, "lorem"));
// return true

try {
  assert1(false);
} catch (e) {
  alert(e);
}
// Uncaught Error: Assertion failed:

try {
  assert1(false, "lorem 1");
} catch (e) {
  alert(e);
}
// Uncaught Error: Assertion failed: lorem 1


/* default export */

import { default as assert2 } from "./assert.esm.js";
window.assert2 = assert2;

alert(alert(assert2(true)));
// return true

alert(assert2(true, "lorem"));
// return true

try {
  assert2(false);
} catch (e) {
  alert(e);
}
// Uncaught Error: Assertion failed:

try {
  assert2(false, "lorem", 3.14, [4,5,6], {a:1,b:2}, 42, "ipsum");
} catch (e) {
  alert(e);
}
// Uncaught Error: Assertion failed: "lorem" 3.14 [4,5,6] {"a":1,"b":2} 42 "ipsum"
